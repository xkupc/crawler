CREATE TABLE `tc_video` (
  `VIDEO_ID` varchar(32) CHARACTER SET utf8 NOT NULL COMMENT '主键',
  `VIDEO_URL` varchar(300) CHARACTER SET utf8 NOT NULL COMMENT '视频地址',
  `VIDEO_NAME` varchar(20) CHARACTER SET utf8 NOT NULL COMMENT '视频名',
  `VIDEO_MARK` varchar(2) CHARACTER SET utf8 DEFAULT NULL COMMENT '观看视频条件1免费2vip3用券4付费',
  `VIDEO_TYPE` varchar(5) CHARACTER SET utf8 DEFAULT NULL COMMENT '视频类型',
  `VIDEO_IMG_URL` varchar(300) CHARACTER SET utf8 DEFAULT NULL COMMENT '视频图片地址',
  `VIDEO_SCORE` varchar(5) CHARACTER SET utf8 DEFAULT NULL COMMENT '视频评分',
  `VIDEO_DESC` varchar(100) CHARACTER SET utf8 DEFAULT NULL COMMENT '视频描述',
  `STATUS` varchar(2) CHARACTER SET utf8 NOT NULL DEFAULT 'Y' COMMENT '默认状态',
  `REMARK` varchar(128) CHARACTER SET utf8 DEFAULT NULL COMMENT '备注',
  `ENABLE_FLAG` char(1) CHARACTER SET utf8 NOT NULL DEFAULT 'Y' COMMENT 'Y启用N禁用',
  `CREATE_BY` varchar(32) CHARACTER SET utf8 DEFAULT NULL COMMENT '创建人',
  `CREATE_DATE` datetime NOT NULL COMMENT '创建时间',
  `LAST_UPDATE_BY` varchar(32) CHARACTER SET utf8 DEFAULT NULL COMMENT '更新人',
  `LAST_UPDATE_DATE` datetime DEFAULT NULL COMMENT '更新时间',
  PRIMARY KEY (`VIDEO_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='腾讯视频';
